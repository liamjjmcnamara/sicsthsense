144 
145  SQLiteDatabase database = this.getReadableDatabase(); //Connect to the database
146 
147  //Simply get all sensors in the configuration table
148  Cursor cursor = database.query(TABLE_SENSOR_CONFIGURATION, columnsToRetrieve, null, null, null, null, null, null);
149  if (cursor.moveToFirst()){ //True if there is at least one result to process
150  do { //Loop through
151  Integer sensorType = cursor.getInt(0);
152  String sensorAddress = cursor.getString(1);
153  Boolean sensorEnabled = cursor.getInt(2)>0;
154  sensorList.add(new SensorInfo(sensorType, sensorAddress, sensorEnabled));
155  } while (cursor.moveToNext());
156  }
157  database.close(); //Close the database connection
158  //Return the list of sensor pairs, or null if none were found
159  return sensorList;
